\item Suppose $f$ is a differentiable function with values as shown. What is $\displaystyle (f^{-1})'(0)$?  

\begin{multicols}{2}
	\begin{tabular}{r|c|c|c|c}
	$x$ & 0 & 1 & 2 & 3 \\
	\hline
	$f(x)$ & 4 & 2 & 0 & -2 \\
	\hline
	$f'(x)$ & $0$ & $-1$ & $-2$ & $-4$ 
	\end{tabular}
	\columnbreak
	\begin{enumerate}\setlength{\itemsep}{.4 cm}
	\item[(A)] $-\frac13$
	\item[(B)] $-\frac12$
	\item[(C)] $-3$
	\item[(D)] $-2$
	\item[(E)] $\frac12$
	\end{enumerate}
\end{multicols}

\item Suppose that $f(x) = \displaystyle \int_{x}^{2} \cos(t^2) \ dt$.  What is $f'(x)$?

\begin{enumerate}
\item[(A)] $\cos(x^2)$
\item[(B)] $\displaystyle \frac{\sin (x^2)}{2x}$
\item[(C)] $-2x\sin(x^2)$
\item[(D)] $-\cos (x^2)$
\item[(E)] None of the above
\end{enumerate}

\end{enumerate}
\end{multicols}
