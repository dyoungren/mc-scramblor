\item Suppose $f'(x) = 2+ \sin(x)$. On which interval below is $f$ increasing?

   \vskip .25 cm

\begin{multicols}{2}
\begin{enumerate}\setlength{\itemsep}{.25cm}
	\item  $\ds (-\infty,\infty)$
	\item  $\ds \left(-\frac\pi2,\frac\pi2\right)$
	\item  $\ds (\pi,2\pi)$
	\item  $\ds [1,3]$
	\item  $f$ is never increasing. 

\end{enumerate}
 \end{multicols}
