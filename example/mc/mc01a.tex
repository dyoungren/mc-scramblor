\item Compute the following limit. \[
			\lim_{h\to 0}\frac{6(1+h)^{4/3}-6}{h}
		\]

   \vskip .25 cm

\begin{multicols}{2}
 \begin{enumerate} \setlength{\itemsep}{.25cm}
	 \item $-\frac{\sqrt{4}}{3}$ % correct
	 \item $8$
	 \item $0$
	 \item $\frac13$
	 \item The limit does not exist.
 \end{enumerate}
\end{multicols}
