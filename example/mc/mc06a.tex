\item Compute the following derivative. \[
	\dfrac{d}{dx}\left(\frac{x}{\sin x}\right)
\] \vskip .25 cm

\begin{multicols}{2}
\begin{enumerate} \setlength{\itemsep}{.25cm}
		\item  $\ds \csc (x)-x \cot (x) \csc (x)$ % correct
	\item    $\ds \sec x$ 
		\item  $\ds \dfrac{\cos x}{\sin^2 x}$
		\item $\ds -\dfrac{x \cos x - \sin x}{\sin^2 x}$		    
		\item $\ds \csc x + x \sec x$
	\end{enumerate}
\end{multicols}
